# setup

```
# brew 
/bin/bash -c z "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"

# when brew installed

brew install nvim
brew install tmux
brew intsall lazygit

# nvm

curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash

```
